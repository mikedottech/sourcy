==== General Compiler info ====

compile.exe krono.cns krono2.cns krono3.cns krono.cmd --output LUA --noopwrap

output:
	crono.lua
	crono2.lua
	crono3.lua
	cronocmd.lua
	cronomain.lua	-> require()s the other files
	
pipeline:
	
	(
		FRONT-END (fixed)
			- parser		(file -> AST)
			- error check	(AST -> AST)
			* (optimize?)	(AST -> AST)
		
		BACK-END (interchangeable)
			- codegen		(AST -> codeblocks)
	)+
	
	- codegen generates one codeblock per info/state block
	- states are checked between files for duplicity
	
	(
		CODE GENERATION
			- Generate code files
			- Generate master file
			- Generate localization/balancing files?
	)

	( (optional)
		ASSEMBLER
			- Generates bytecode files (luac?)		
	)?
